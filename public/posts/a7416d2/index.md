```
---
title: 
subtitle:
date: 2025-12-17T20:55:39+08:00
draft: true
description:
keywords:
license:
comment: false
weight: 0
tags:
  - 
categories:
  - 
collections:
 - 
hiddenFromHomePage: false
hiddenFromSearch: false
hiddenFromRelated: false
hiddenFromFeed: false
summary:
resources:
  - name: featured-image
    src: featured-image.jpg
  - name: featured-image-preview
    src: featured-image-preview.jpg
toc: true
math: false
lightgallery: false
password:
message:
repost:
  enable: true
  url:
---

```
